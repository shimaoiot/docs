    # 解绑设备

    上行/下行

    发起方：客户端/服务端

    协议：MQTT

    method：device-unbind

    上行 Topic：/topic/pub/device/{AppKey}/{clientId}

    下行 Topic：/topic/sub/device/{AppKey}/{clientId}

    # header 信息

    header 信息包含所有请求必须的通用信息

    # 示例
    {
  "header": {
    "traceId": "xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx",
    "payloadVersion": 1,
    "brand": "KT",
    "timestamp": 1234567890123,
    "method": "device-unbind"
  }
}

    # 请求参数说明
    参数名 参数类型 参数说明 是否必填
    gatewayId String 网关设备唯一ID 是
    devices JSON Object Array 设备列表 是
    └deviceCode String 子设备唯一 ID 是
    └deviceName String 设备名称 否
    └deviceType String 设备类型 否
    # 请求参数示例
    {
  "header": {
    "traceId": "xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx",
    "payloadVersion": 1,
    "brand": "KT",
    "method": "device-unbind",
    "timestamp": 1234567890123
  },
  "payload": {
    "gatewayId": "aaabbbccc",
    "devices": [
      {
        "deviceCode": "12345",
        "deviceName": " 灯 ",
        "deviceType": "light"
      },
      {
        "deviceCode": "ABCDEF",
        "deviceName": " 灯 ",
        "deviceType": "light"
      }
    ]
  }
}
